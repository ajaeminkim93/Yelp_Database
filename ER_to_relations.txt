CREATE TABLE Business(
	business_id VARCHAR(30),
	review_count INTEGER,
	stars FLOAT,
	latitude FLOAT,
	name VARCHAR(30),
	street_address VARCHAR(40),
	city VARCHAR(30),
	state VARCHAR(30),
	postal_code INTEGER,
	number of checkins INTEGER, //function to calculate this?
	PRIMARY KEY(business_id),
)

CREATE TABLE Checkin(
	time TIME,
	date DATE,
	business_id VARCHAR(30),
	PRIMARY KEY(business_id, time, date),
	CONSTRAINT 'businessID' FOREIGN KEY(business_id) REFERENCES Business(business_id),
)

CREATE TABLE Tip(
	text VARCHAR(200),
	date DATE,
	time TIME,
	likes INTEGER, //function?
	business_id VARCHAR(30),
	user_id VARCHAR(30),
	PRIMARY KEY(business_id, user_id, date, time),
	CONSTRAINT 'businessID' FOREIGN KEY(business_id) REFERENCES Business(business_id),
	CONSTRAINT 'userID' FOREIGN KEY(user_id) REFERENCES User(user_id),
)

CREATE TABLE User(
	user_id VARCHAR(30),
	yelping_since DATE,
	longitude FLOAT, //function? user entered
	latitude FLOAT, //function? user entered
	useful INTEGER, //function derived
	tip count INTEGER, //function derived
	total tip likes INTEGER, //function derived
	fans INTEGER, //function derived
	count of votes INTEGER, //function derived
	average stars FLOAT, //function derived
	PRIMARY KEY(user_id),
)

CREATE TABLE near(
	business_id VARCHAR(30),
	user_id VARCHAR(30),
	nearest FLOAT //function derived
	PRIMARY KEY(business_id, user_id),
	CONSTRAINT 'businessID' FOREIGN KEY(business_id) REFERENCES Business(business_id),
	CONSTRAINT 'userID' FOREIGN KEY(user_id) REFERENCES User(user_id),
)
